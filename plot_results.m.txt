function plot_results(time,state)

u=state(1,:);
theta=state(8,:);
x=state(10,:);
y=state(11,:);
z=state(12,:);

figure

subplot(3,1,1)
plot(time,u,'LineWidth',2)
title('Velocity')

subplot(3,1,2)
plot(time,theta,'LineWidth',2)
title('Pitch Angle')

subplot(3,1,3)
plot3(x,y,z,'LineWidth',2)
grid on
title('3D Trajectory')

end
